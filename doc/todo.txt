Permissions:
ABAC (Attribute Based Access Control)

Connection ->
  - None Permission is granted ()
  - auth is required for more / different access


// Permission Tree:
//  If no leaf: use node above: Value = Option<T>
//
//  Operation
//   - Write
//   - Read
//   - Transaction
//  Permission
//   - auth (for "default" must be yes)
//

permissions:
  root:
    - 
      authtoken: str

  admin:
    - 
      authtoken: str
      operations:
        - *string: bool  (can be any number of strings, things like write.put: true, read: false)

  user1
    - 
      authtoken: str
      operations:
        - *string: bool
